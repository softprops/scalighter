abstract class Boom[J[_]] extends Foo[String <% J, Int] {
  def cond: Boolean = try {
    1 * 2 == 3 && 2 * 4 == 6 || 2 / 4 == 2
  } catch { case e @ Exception => true

  def apply[A <: AnyRef](xs: A*): Array[A] = {
    val array = new Array[A](xs.length)
    var i = 0
    for (x <- xs.elements) { array(i) = x; i += 1 }
    array
  }

  def m(x: t forSome {type t <: A}) = x

  def foreva = while(cond) println("4eva")

  def goodEnough[T](t: T): Boolean

  def `package`[T](x: T) =
   if(goodEnough(x))
   else {
     `package`(x)
   }
